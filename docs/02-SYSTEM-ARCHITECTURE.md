    def initialize_pool(self, plc_code, ip, port, pool_size=5):
        """
        PLC ì—°ê²° í’€ ì´ˆê¸°í™”

        Args:
            plc_code (str): PLC ì½”ë“œ (ì˜ˆ: 'PLC01')
            ip (str): PLC IP ì£¼ì†Œ

                'ip': ip,
                'port': port
            }

                    conn.setaccessopt(commtype="ascii")
                    pool.put(conn)
                    self.logger.comm.info(f"Connection {i+1}/{pool_size} created for {plc_code}")

        """
        ì—°ê²° ê°€ì ¸ì˜¤ê¸° (ë¸”ë¡œí‚¹)

        Args:
            plc_code (str): PLC ì½”ë“œ
            timeout (float): ëŒ€ê¸° ì‹œê°„ (ì´ˆ)

        Returns:
            pymcprotocol.Type3E: PLC ì—°ê²° ê°ì²´

            Empty: íƒ€ì„ì•„ì›ƒ
        """
        pool = self._pools.get(plc_code)

            raise ValueError(f"Connection pool for {plc_code} not initialized")

            conn = pool.get(timeout=timeout)

                config = self._configs[plc_code]
                conn = pymcprotocol.Type3E()
                conn.connect(config['ip'], config['port'])
                conn.setaccessopt(commtype="ascii")

            return conn

        except Empty:
            self.logger.error.error(f"Timeout acquiring connection for {plc_code}")
            raise

    def release(self, plc_code, conn):
        """
        ì—°ê²° ë°˜í™˜

        Args:
            plc_code (str): PLC ì½”ë“œ
            conn: PLC ì—°ê²° ê°ì²´
        """
        pool = self._pools.get(plc_code)

                        conn.close()
                    except:
                        pass
